	&&\\
	\multicolumn{3}{l}{\textbf{Linear Regression:}} \\
	J(\theta) & = & \frac{1}{2m}(X\theta - y)^T(X\theta - y) + \frac{\lambda}{2m}[(L\theta)^T(L\theta)]\\
	&&\\
	\nabla J   & = & \frac{1}{m}(X^T(X\theta - y) + \lambda L\theta) \\
	&&\\
	\theta & := & (1-\frac{\alpha\lambda}{m}L\theta) - \frac{\alpha}{m}X^T(X\theta - y)\\
	&&\\
	\multicolumn{3}{l}{\textbf{Logistic Regression:}} \\
	h & = & \sigma(X\theta) = \frac{1}{1-e^{-X\theta}} \\
	&&\\
	J(\theta) & = & -\frac{1}{m}[y^T\log(h) + (1-y)^T\log(1-h)] + \frac{\lambda}{2m}[(L\theta)^T(L\theta)]\\
	&&\\
	\nabla J   & = & \frac{1}{m}(X^T(h-y) + \lambda L\theta) \\
	&&\\
	\theta & := & (1-\frac{\alpha\lambda}{m}L\theta) - \frac{\alpha}{m}X^T(h - y)\\
